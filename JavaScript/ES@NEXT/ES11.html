# BigInt

新的原始数据类型

const max = 2 ** 53; //9007199254740992
const max_safe = Number.MAX_SAFE_INTEGER; //9007199254740991

// 到顶了，所以再加也是最大值
console(max === max + 1); //true
// 实际输出9007199254740992
console.log(9007199254740993);

// 使用bigInt两种方式：

// 第一种结尾加n
console.log(9007199254740993n); // 9007199254740993n
// 第二种BigInt()
const num = BigInt(9007199254740993n);
// 可以比较数值
console.log(1n == 1); //true
// == 只比较数值，===比较数值以及类型
console.log(typeof num); //bigint
console.log(1n === 1); //false

// 如果不想要结尾的n,可以使用toString方法，存储到数据库中只能以字符串的形式可以存储
console.log(num.toString()); //9007199254740993

// 计算,bigInt类型才有效
console.log(9007199254740993n + 1n); //9007199254740994n

// 推荐做法：结尾都加n，不然可能会出现以下情况
console.log(BigInt(9007199254740995)); // 9007199254740996n
console.log(BigInt(9007199254740995) === BigInt(9007199254740996n)); // true

# Promise.allSettled

Promise 扩展

对于并发任务的处理，都可以得到对应的状态

Promise.all([
  Promise.resolve({
    code: 200,
    data: [1, 2, 3],
  }),
  Promise.resolve({
    code: 200,
    data: [4, 5, 6],
  }),
  Promise.resolve({
    code: 200,
    data: [7, 8, 9],
  }),
])
  .then((res) => {
    console.log("success", res);

    // success [
    //     { code: 200, data: [ 1, 2, 3 ] },
    //     { code: 200, data: [ 4, 5, 6 ] },
    //     { code: 200, data: [ 7, 8, 9 ] }
    //   ]
  })
  .catch((err) => {
    console.log("fail", err);
  });

// 有一个失败，就都拿不到了
Promise.all([
  Promise.resolve({
    code: 200,
    data: [1, 2, 3],
  }),
  Promise.reject({
    code: 500,
    msg: "error",
  }),
  Promise.resolve({
    code: 200,
    data: [7, 8, 9],
  }),
])
  .then((res) => {
    console.log("success", res);
  })
  .catch((err) => {
    console.log("fail", err);
    // fail { code: 500, msg: 'error' }
  });

Promise.allSettled([
  Promise.resolve({
    code: 200,
    data: [1, 2, 3],
  }),
  Promise.reject({
    code: 500,
    msg: "error",
  }),
  Promise.resolve({
    code: 200,
    data: [7, 8, 9],
  }),
])
  .then((res) => {
    console.log("success", res);
    // success [
    //     { status: 'fulfilled', value: { code: 200, data: [Array] } },
    //     { status: 'rejected', reason: { code: 500, msg: 'error' } },
    //     { status: 'fulfilled', value: { code: 200, data: [Array] } }
    //   ]
    const data = res.filter((item) => item.status === "fulfilled");
  })
  .catch((err) => {
    console.log("fail", err);
  });

# 全局对象

提供了一个标准的方式去获取不同环境下的全局对象

web：window/self

node：global

// old：通过封装方法判断获取全局对象
function getGlobal() {
  if (typeof self !== "undefined") {
    return self;
  }
  if (typeof window !== "undefined") {
    return window;
  }
  if (typeof global !== "undefined") {
    return global;
  }
  throw new Error("无法找到全局对象");
}

console.log(getGlobal());

// new:直接使用globalThis
console.log(globalThis);

# 可选链

场景：对象具有多层嵌套的时候，我们想访问内层对象的属性或方法

注意： ?.是一体的，如果拆开会出现语法错误

const user = {
  userInfo: {
    name: "Monster",
    getAge() {
      return 18;
    },
  },
};

// old:通过&号一层一层判断很麻烦
const name = user && user.userInfo && user.userInfo.name;
const age =
  user && user.userInfo && user.userInfo.getAge && user.userInfo.getAge();

// new:使用?.判断这个对象的属性或方法是否存在，如果存在再用.去调用
const name = user?.userInfo?.name;
const age = user?.userInfo?.getAge?.();

# 空值合并运算符

为了解决|带来的问题：

方式：??

只有值为 null，undefined 的时候才取默认值

// const b = 2;
// const b = 0;
// const b = "";
// const b = false;
// const b = null;
// const b = undefined;

const a = b || 5;

console.log(a);
// 2
// 5
// 5
// 5
// 5
// 5

// 问题：可是我的需求是输入0,false,''也需要对应输出,但是实际却不是的，而且我们大多数希望值为null或undefined的时候，才去除默认值，否则应该取b,

// const b = 2;
// const b = 0;
// const b = "";
// const b = false;
// const b = null;
// const b = undefined;

const a = b ?? 5;

console.log(a);
// 2
// 0
//
// false
// 5
// 5
